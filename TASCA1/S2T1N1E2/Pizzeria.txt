//Llista quants productes de categoria 'Begudes' s'han venut en una determinada localitat.

select count(begudes.idbegudes), localitat.nom, begudes.nom from begudes
inner join comanda on begudes.comanda_idcomanda = comanda.idcomanda
inner join tenda on comanda.tenda_idtenda = tenda.idtenda
inner join localitat on tenda.idtenda = localitat.idLocalitat
where idlocalitat = 1

//Llista quantes comandes ha efectuat un determinat empleat/da
select count(comanda.idcomanda), treballador.nom from comanda
inner join tenda on comanda.tenda_idtenda = tenda.idtenda
inner join treballador on treballador.idtreballador
